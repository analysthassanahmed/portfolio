#function returns the status of the match relative to home team (win,lose,draw)
def status():
  global win_list  # declare win_list as global
  win_list = []
  for i in df.goal_diff:
    if i < 0:
      status = "lose"
    elif i == 0:
      status = "draw"
    else:
      status = "win"
    win_list.append(status)# Add status to list "win_list"

# Call the function to populate win_list
status()
df["status"] = win_list#make new column with the list
